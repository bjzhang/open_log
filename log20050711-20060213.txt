	while(1){
		if(RC500USB_request (0,type)!=0) //对第一个读卡器进行操作	//zj:返回0表示成功
			continue;
		if(RC500USB_anticoll(0,snr)!=0) //进行防碰撞选择成功则返回卡号
			continue;
		if(RC500USB_select(snr,size)!=0) //选择卡
			continue;
		if(RC500USB_authkey(0,szpwd,0)!=0) //验证第0 扇区的密钥A
			continue;
		if(RC500USB_write(2,data)!=0) //往第二块写入数据
			continue;
		if(RC500USB_read(2,data)!=0) //读取第二块的数据
			continue;
		for(i=0;i<16;i++){
			printf("%x ",data[i]);
		}
		printf("\n");
		if(RC500USB_writeval(1,value)!=0) //往第一块写入数值value
			continue;
		if(RC500USB_value (0xc1,1,value,1)!=0) //把第一块的数值加上value
			continue;
		if(RC500USB_readval(1,value)!=0) //读取第一块的数值
			continue;
		printf("%x\n",value);
		RC500USB_halt(); //使卡进入halt 状态
		RC500USB_buzzer(1,25,25,1); //让蜂鸣器响一次
		//sleep(50);
	}
注意
1，“块”的计算：在MUR-100中，块是不分扇区（sector）统一编号的。
例如，第5扇区第1块为：5*4+1=21。
2，选择扇区的方法：利用RC500USB_authkey(密钥类型,密钥值,扇区号)，若验证通过，则选中该扇区。例如：RC500USB_authkey(0,szpwd,0)，验证第0扇区A密钥是否为szpwd，若相符则选中0扇区。

#pragma vector=PORT2_VECTOR
__interrupt void key_isr()
{
    
    char line_mask,i;
    char line,val,key_value;
    
    val=P2IFG&0x0f;  //从中断标志寄存器看列值
    switch(val)
    {
        case 0x01:
            val=1;
        break;
        case 0x02:
            val=2;
        break;
        case 0x04:
            val=3;
        break;
        case 0x08:
            val=4;
        break;
        default:
            val=0;
        break;
    }
    //测试行值，将4根行线依次置高，检查4根列线的值，以此判断行值
    for(line=7,i=0;i<4;i++)
    {
       line_mask=0x10<<i;
       P2OUT&=0x0f;
       P2OUT|=line_mask;
       if((P2IN&0x0f)==0x0f)
       {
           line=i;
           break;
       }
    }
    //show_num(0,val);  //测试用的两行
    //show_num(1,line);
    
    
    //计算key_value
    if(line<3&&val<4)
        key_value=(2-line)*3+val; //zj:line取0，1，2，值越大按键对应数字越小，所以用2-line
    else if(line==3)
    {
         switch(val)
         {
              case 1:
                  key_value=0;
              break;
              case 2:
                   key_value=DOT;
              break;
              case 3:
                   key_value=ORIGINAL_PRESS;
              break;
              case 4:
                   key_value=85;
              break;
              default:
                   break;
         }
    }
    else
    {
          switch(line)
          {
               case 0:
                     key_value=152;
               break;
               case 1:
                     key_value=122;
               break;
               case 2:
                     key_value=100;
               break;
               default:
               break;
          }
    }
    key_function(key_value);
   
    P2OUT&=0x0f; //把输出的高4位清零
    P2IFG=0x00; // 清中断标志位
}

1，系统移植分为版级，芯片剂和cpu级（好像是）三级，难度一次增大。
2，系统移植的目的：使已有系统运行于目标系统上。

1，在一个已经有现成实验和系统已经支持的特定的板子上学习系统移植，主要是学习系统移植的基本思路。并且能够按需进行剪裁。
2，理解该操作系统对板子的改动的原因。
3，在此基础上在一个新系统上进行对同一操作系统移植。
4，在新板上对新操作系统进行移植。
5，如果是软硬件协同设计，可以根据需要决定是由软件实现还是由硬件实现。并且能判断哪个层次的软件实现（操作系统，驱动，应用软件等等）。

资料，c语言，自-C语言高效编程的几招 ++
使用位操作
　　实现高效的C语言编写的第三招——使用位操作，减少除法和取模的运算。
　　在计算机程序中，数据的位是可以操作的最小数据单位，理论上可以用“位运算”来完成所有的运算和操作。一般的位操作是用来控制硬件的，或者做数据变换使用，但是，灵活的位操作可以有效地提高程序运行的效率。举例如下：
方法G
int I,J;
I = 257 /8;
J = 456 % 32;
方法H
int I,J;
I = 257 >>3;
J = 456 - (456 >> 4 << 4);
　　在字面上好像H比G麻烦了好多，但是，仔细查看产生的汇编代码就会明白，方法G调用了基本的取模函数和除法函数，既有函数调用，还有很多汇编代码和寄存器参与运算；而方法H则仅仅是几句相关的汇编，代码更简洁，效率更高。当然，由于编译器的不同，可能效率的差距不大，但是，以我目前遇到的MS C ,ARM C 来看，效率的差距还是不小。相关汇编代码就不在这里列举了。
运用这招需要注意的是，因为CPU的不同而产生的问题。比如说，在PC上用这招编写的程序，并在PC上调试通过，在移植到一个16位机平台上的时候，可能会产生代码隐患。所以只有在一定技术进阶的基础下才可以使用这招。

dsp作业，h.264，计划
1，3月初-3月15日之前，争取实现变换和量化的代码。如果3月底仍不能实现，就放弃h.264计划，改为实现一个图像压缩的程序。
2，3.15-4.15，h.264代码整体移植。采取自动移植的办法：搜索代码中dsp无法实现的部分，用一个的数据结构实现数据，用四则运算函数实现运算，例如浮点数a.b，定义为
//A=a, 	B=0.b*10^n, 0.b*10^n中的n为使0.b*10^n为最小整数的n。例如12.56, A=12, B=56
struct floatType{
integerType A;
decimalType B;
};
设a.b*c.d=e.f
a.b*c.d=(a+0.b)(c+0.d)=a*c+a*0.d+0.b*c+0.b*0.d=
A*C+A*D/(D/0.d)+B*C/(B/0.b)+B*D/(B/0.b*D/0.d)
所以，e=A*C+A*D/(D/0.d)+B*C/(B/0.b)+B*D/(B/0.b*D/0.d) 
      f=A*D%(D/0.d)+B*C%(B/0.b)+B*D%(B/0.b*D/0.d)
其中，D/0.d表示上面的10^n，例如12.56中，A=12, b=56, B/0.b=100
其他运算同理。如果这个运算经过比较大的优化，凭着blackfin的优秀运算单元，性能降低不会太多。
这种方法的好处是，快速向没有浮点运算单元的dsp移植程序。如果程序不常用，所需MIPS少，而且浮点运算也比较少，这种移植性能降低应该不多，可以提高开放效率。而且为开发提供原形，可以很快就在目标板上进行测试，而且便于比较是否适用浮点运算对性能的影响。
刚才搜索了一下，发现浮点运算很少，只在两个文件中出现。如果这种方法不行，直接修改应改也可以。
